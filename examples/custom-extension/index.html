📺 自定义节点：YouTube 嵌入

创建一个可以嵌入 YouTube 视频的自定义节点。

// 1. 定义节点
const YoutubeNode = Node.create({
  name: 'youtube',
  group: 'block',
  atom: true,
  
  addAttributes() {
    return {
      src: { default: null },
      width: { default: 640 },
      height: { default: 360 },
    };
  },
  
  parseHTML() {
    return [{ tag: 'div[data-youtube]' }];
  },
  
  renderHTML({ HTMLAttributes }) {
    return ['div', { 'data-youtube': '', ...HTMLAttributes }];
  },
  
  addNodeView() {
    return ({ node }) => {
      const div = document.createElement('div');
      div.className = 'youtube-embed';
      div.innerHTML = \`\`;
      return { dom: div };
    };
  },
});

// 2. 添加命令
addCommands() {
  return {
    setYoutube: (options) => ({ commands }) => {
      return commands.insertContent({
        type: 'youtube',
        attrs: options,
      });
    },
  };
}

🧮 自定义标记：数学公式

创建一个用于显示数学公式的行内标记。

// 定义行内数学标记
const MathInline = Mark.create({
  name: 'mathInline',
  
  addAttributes() {
    return {
      formula: { default: '' },
    };
  },
  
  parseHTML() {
    return [
      {
        tag: 'span[data-math]',
        getAttrs: element => ({
          formula: element.getAttribute('data-math'),
        }),
      },
    ];
  },
  
  renderHTML({ HTMLAttributes }) {
    return ['span', {
      'data-math': HTMLAttributes.formula,
      class: 'math-inline',
    }, HTMLAttributes.formula];
  },
  
  addCommands() {
    return {
      setMathInline: (formula) => ({ commands }) => {
        return commands.setMark('mathInline', { formula });
      },
      unsetMathInline: () => ({ commands }) => {
        return commands.unsetMark('mathInline');
      },
    };
  },
});

🎨 自定义工具栏组件

创建可复用的自定义工具栏按钮和下拉菜单。

// Emoji 选择器组件
function EmojiPicker({ onSelect }) {
  const emojis = ['😀', '😂', '🥰', '😎', '🤔', '👍', '🔥', '💯'];
  
  return {
    render() {
      const div = document.createElement('div');
      div.className = 'emoji-picker';
      emojis.forEach(emoji => {
        const btn = document.createElement('button');
        btn.textContent = emoji;
        btn.onclick = () => onSelect(emoji);
        div.appendChild(btn);
      });
      return div;
    }
  };
}

// 颜色选择器组件
function ColorPicker({ colors, onSelect }) {
  return {
    render() {
      const wrapper = document.createElement('div');
      wrapper.className = 'color-picker';
      
      colors.forEach(color => {
        const swatch = document.createElement('button');
        swatch.className = 'color-swatch';
        swatch.style.backgroundColor = color;
        swatch.onclick = () => onSelect(color);
        wrapper.appendChild(swatch);
      });
      
      return wrapper;
    }
  };
}

// 扩展开发最佳实践

// 1. 命名规范
// - 节点使用 PascalCase: YoutubeNode, CalloutNode
// - 标记使用 camelCase: mathInline, textColor
// - 命令使用 camelCase: setYoutube, toggleCallout

// 2. 属性定义
addAttributes() {
  return {
    // 使用 default 指定默认值
    src: {
      default: null,
      // 使用 parseHTML 从 HTML 解析
      parseHTML: element => element.getAttribute('src'),
      // 使用 renderHTML 渲染到 HTML
      renderHTML: attributes => ({ src: attributes.src }),
    },
  };
}

// 3. 输入规则（自动转换）
addInputRules() {
  return [
    // 输入 :emoji: 自动转换
    nodeInputRule({
      find: /:([a-z]+):$/,
      type: this.type,
      getAttributes: match => ({ emoji: match[1] }),
    }),
  ];
}

// 4. 粘贴规则
addPasteRules() {
  return [
    nodePasteRule({
      find: /https:\/\/youtube\.com\/watch\?v=([a-zA-Z0-9_-]+)/g,
      type: this.type,
      getAttributes: match => ({ 
        src: `https://www.youtube.com/embed/${match[1]}` 
      }),
    }),
  ];
}

// 5. ProseMirror 插件
addProseMirrorPlugins() {
  return [
    new Plugin({
      props: {
        handleKeyDown(view, event) {
          // 自定义键盘处理
          return false;
        },
      },
    }),
  ];
}
